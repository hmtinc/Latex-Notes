\section{Probability Rules and Conditional Probability}

\subsection{Relations from Set Theory}
\begin{itemize}
\item The \textbf{Union} of two events A and B (\(A \cup B\)) is the set consisting of all outcomes that are either in A, in B, or in both events
\item The \textbf{Intersection} of two events A and B (\(A \cap B\)) is the set of all outcomes in both A and in B 
\item The \textbf{Complement} of an event A denoted by \(\bar{A}, A^c\), is the set of all outcomes that are not contained in A  
\item The \textbf{Empty Event} or the \textbf{Null Set} \(\varnothing\)
\begin{itemize}
\item \(\varnothing = \bar{S}\)
\item \(P(\varnothing) = 0\)
\item \(P(S) = 1\)
\end{itemize}
\end{itemize}

\subsection{De Morgan's Laws}
Union and intersection interchange under complementation 
\begin{itemize}
\item \(\bar{A \cup B} = \bar{A} \cap \bar{B}\)
\item \(\bar{A \cap B} = \bar{A} \cup \bar{B}\)
\end{itemize}

More Generally Defined as : 
\begin{itemize}
\item \(\bar{A_1 \cup A_2 \cup \ldots \cup A_k} = \bar{A_1} \cap \bar{A_2} \cap \ldots \cap \bar{A_k}\)
\item \(\bar{A_1 \cap A_2 \cap \ldots \cap A_k} = \bar{A_1} \cup \bar{A_2} \cup \ldots  \cup\bar{A_k}\)
\end{itemize}

\subsection{Rules for Unions of Events : Addition Law of Probability of the Sum Rule}

\subsubsection{Mutually Exclusive Events}
\begin{itemize}
\item The Probability of either event happening is the sum of their individual probabilities
\item \(P(A \cup A_2 \cup \ldots \cup A_k) = P(A_1) + P(A_2) + \ldots + P(A_K)\)
\end{itemize}
