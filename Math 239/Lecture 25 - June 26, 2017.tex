\section{BFS Properties}
Let (T,p) be a BFS tree with roots and let Level u be the vertices x such that the xr-path in T has length i.
\begin{enumerate}
\item G is connected \(\iff\) V(T) = V(G)
\item P is a shortest xr-path in G 
\item G has an odd cycle \(\iff\) exists \(x y \in E(G) \bs E(T)\) with level(x)=level(y)
\end{enumerate}

\section{Planar Graphs}

\begin{definition}
A graph is \textbf{Planar} if it has a drawing in the plane in which every two edges intersect only at their ends.

Any such drawings (without crossings) is a planar embedding
\end{definition}

\begin{definition}
A \textbf{face} is a region of a drawing
\end{definition}


\begin{definition}
Given a face of a planar embedding, a \textbf{Boundary walk}.
\(W_f = v_0, e_1v_1, \ldots, v_{n-1}e_n v_n\) is obtained by following the perimeter of a boundary of f.
\end{definition}

\textbf{Note : } The degree of f (denoted deg(f)) is the length of f. 

\begin{lemma}
\textbf{Faceshaking Lemma :} Let F be the set of faces in a planar embedding of a connected graph. Then, 
$$\sum_{f \in F} deg(f) = 2 \mid E(G) \mid $$
\end{lemma}

\begin{proof}
Each edge e has two sides (left and right side) 

\begin{exblock}{Fact}
Both sides of an edge e belong to the same face \(\iff\) e is a bridge
\end{exblock}


If both sides of e are on the same face, e contributes in 2 to the degree of such face\\
If the sides are on distinct faces, then e contributes in 1 to the degree of these faces.

Then, \(\sum deg(f) = 2 \mid E(G) \mid \)

\end{proof}
