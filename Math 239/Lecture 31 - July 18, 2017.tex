\section{Matchings}
Image goes here 

\begin{definition}
Given a graph G a matching is  set M of edges of G such that every vertex in G is incident to at most one edge in M
\end{definition}

\begin{definition}
Let M be a matching of G 
\begin{enumerate}
\item M is \textbf{Maximum} if every other matching \(M^\prime\) has \(\mid M^\prime \mid \in \mid M \mid \)\\
(\textcolor{red}{Red edges} form a maximum matching in \(G_1\) and \(G_2\) but not in \(G_3\))
\item A matching is \textbf{perfect} if \(\mid M \mid = \frac{1}{2} \mid V(G) \mid \)\\
(\textcolor{red}{Red edges} form a perfect matching just for \(G_1\))
\item A vertex \(v \in V(G)\) is \textbf{saturated} by M if it is incident to an edge in M. 
\end{enumerate}
\end{definition}

\section{Augmenting paths}
\textbf{Goal:} Given a graph, find a maximum matching. 

\begin{itemize}
\item How to tell that a matching is not maximum? \textcolor{red}{Use Augmenting paths} 
\item How to tell a graph is not connected? \textcolor{red}{There exists a cut}
\item How to tell a graph is bipartite? \textcolor{red}{Check for Odd cycles}
\item How to tell that a matching is not planar \textcolor{red}{Check for \(K_{3,3}\) or \(K_5\) subdivisions}
\end{itemize}

\begin{definition}
Given a graph G and a matching M a path P is \textbf{alternating} if when we traverse \(P\), the edges alternate between \(M\) and \(E(G) \bs M\)


If in addition the ends of \(P\) are unsaturated, then \(P\) is \textbf{Augmenting}
\end{definition}
