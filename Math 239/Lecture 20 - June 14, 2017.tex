\begin{definition}
A graph is \textbf{Connected}, id for every two vertices \(x, y \in V(G)\) there is a path from x to y.
\end{definition}

\begin{theorem}
Let G be a graph and \(v \in V(G)\). G is connected if and only if for every \(w \in V(G)\) there is a path from v to w in G.
\end{theorem}

\begin{proof}
Prove as an exercise. \\
\textbf{Hint :} Suppose that for very \(w \in V(G)\), there is a path from v to w 
\end{proof}

\begin{expx}
Show that the k-cube \(Q_k\) is connected. \\
\textbf{Solution :} By previous theorem, we only need to show that every \(s \in V(Q_k)\) is connected (by using a path) to \(00 \ldots 0\),  So suppose s has \(\ell\) 1s. 
\begin{itemize}
\item \(S_\ell = S\)
\item \(S_{\ell-1} = \) the string obtained from \(S_\ell\) by replacing the first 1 in \(S_\ell\) with
 a 0
\item [\(\vdots\)]
\item  \(S_{0} = 00 \ldots 0\)
\end{itemize}
\(P : S_\ell S_{\ell -1} \ldots S_0\) connects S to \(00 \ldots 0\)
\end{expx}

\begin{definition}
Given a \(G = (V,E)\), a \textbf{Subgraph} \(H = (V^\prime, E^\prime)\) is a graph such that \(V^\prime \subseteq V \) and \(E^\prime \subseteq E\)
\end{definition}

\begin{definition}
A \textbf{Component} of G is a sub-graph \(H\) such that 
\begin{itemize}
\item [(a)] \(H\) is connected 
\item [(b)] \(H\) is not a proper subgraph of a connected subgraph. 
\end{itemize}
\end{definition}

Given \(u, v \in V(G)\) , if we define \(u \sim v\) if u and v are in the same component then 
\begin{itemize}
\item \(a \sim a\)
\item If \(a \sim b \implies b \sim a\)
\item \(a \sim b\) and \(b \sim c \implies a \sim c\) 
\end{itemize}
